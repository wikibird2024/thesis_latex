% Set up the document class
\documentclass[12pt, a4paper]{book}

% Include packages and configurations from the preamble file
\input{preamble.tex}

% Begin the document
\begin{document}

% ---------------------------------------------------------------------
%                          FRONT MATTER
% ---------------------------------------------------------------------
% The title page must be the very first element of the document.
\input{frontmatter/titlepage.tex}
\input{frontmatter/_trang_nhiem_vu.tex}
\input{frontmatter/abstract.tex}

% Set page numbering to Roman numerals (i, ii, iii, ...)
% This should be done after the title pages and before the TOC.
\pagestyle{plain}
\pagenumbering{roman}
\setcounter{page}{1}

% The acknowledgements and table of contents should come after the title pages.
\input{frontmatter/acknowledgements.tex}

% Generate the table of contents and lists
\clearpage
\tableofcontents
\clearpage

\listoffigures
\clearpage

\listoftables
\clearpage

\listoflistings
\clearpage

% ---------------------------------------------------------------------
%                          MAIN MATTER
% ---------------------------------------------------------------------
% Start page numbering with Arabic numerals (1, 2, 3, ...) and reset to 1
% This should be done right before the first chapter.
\pagestyle{fancy}
\pagenumbering{arabic}
\setcounter{page}{1}

% Include the chapters from the "chapters" directory
% Each chapter is in its own file for easy management
\input{chapters/ch1_introduction/chapter1.tex}
\input{chapters/ch2_background/chapter2.tex}
\input{chapters/ch3_methodology/chapter3.tex}
\input{chapters/ch4_results/chapter4.tex}
\input{chapters/ch5_conclusion/chapter5.tex}

% ---------------------------------------------------------------------
%                          BACK MATTER
% ---------------------------------------------------------------------
% Add the appendix
\clearpage
\appendix
\chapter{Phụ lục}
\input{backmatter/appendix_A_code.tex}

% Create the bibliography page
\clearpage
\addcontentsline{toc}{chapter}{Tài liệu tham khảo}
\printbibliography[heading=bibintoc]

% End the document
\end{document}
